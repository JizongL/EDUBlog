## For loop


using `size_t` from the `cstddef` library

```c++
#include <cstddef>
#include <iostream>
 
int main()
{
    const std::size_t N = 10;
    int* a = new int[N];
 
    for (std::size_t n = 0; n < N; ++n)
        a[n] = n+1;
    for (std::size_t n = N; n-- > 0;) // Reverse cycles are tricky for unsigned types.
        std::cout << a[n] << " ";
 
    delete[] a;
}
```
